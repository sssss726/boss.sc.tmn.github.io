<script>
    // Переменная для хранения текущего QR-кода
    let qr;

    function generateQRCode() {
        const input = document.getElementById('inputText').value.trim();
        if (!input) {
            alert('Пожалуйста, введите текст или ссылку');
            return;
        }

        // Получить текущую дату и время
        const now = new Date();
        const dateStr = now.toLocaleString('ru-RU'); // формат по-русски

        // Объединяем введённый текст и дату/время
        const textForQR = `${input}\n\nСоздано: ${dateStr}`;

        // Очистить предыдущий QR-код
        if (qr) {
            qr.clear();
            qr.remove();
        }

        // Создать новый QR-код
        qr = new QRCode(document.getElementById("qrcode"), {
            text: textForQR,
            width: 256,
            height: 256,
            colorDark : "#000000",
            colorLight : "#ffffff",
            correctLevel : QRCode.CorrectLevel.H
        });
    }
</script>
